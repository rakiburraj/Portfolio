CREATE TABLE Patients(
PatientID int primary key,
Name varchar(50),
Age int,
Gender varchar(10),
Address varchar(100),
Phone int,
AdmissionDate Date
);

CREATE TABLE Billing(
BillID int primary key,
Amount int,
Status varchar(20),
BillingDate Date,
PatientID int
Foreign key(PatientID) references Patients(PatientID)
);

CREATE TABLE Insurance(
InsuranceID int primary key,
Provider varchar(50),
PolicyNumber int,
CoverageAmount int,
ExpirationDate Date,
PatientID int
Foreign key (PatientID) references Patients(PatientID)
);
